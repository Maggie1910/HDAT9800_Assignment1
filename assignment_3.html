convert_to_factor <- function(df, cols) {
  for (col in cols) {
    if (col %in% names(df)) {
      df[[col]] <- as.factor(df[[col]])
      print(paste("Column", col, "is converted to factor"))
    } else {
      print(paste("Column", col, "does not exist in the dataframe"))
    }
  }
  return(df)
}
